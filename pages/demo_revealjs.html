Demo2 - show in slide Mode

File: Sample_demo1.pptx

Description:

View all slides of "Sample_demo1.pptx" file in slide Mode.

Code:

              
<div id="slide-resolte-contaniner" ></div> 
<script> 
  $("#slide-resolte-contaniner").pptxToHtml({ 
    pptxFileUrl: "Sample_demo1.pptx", 
    slideMode: true,
    slidesScale: "55%",
    keyBoardShortCut: true, 
    mediaProcess: true, /** true,false: if true then process video and audio files */ 
    slideType: "revealjs",
    revealjsConfig: { /*TODO (https://revealjs.com) */
	transition: "zoom"
     }
  }); 
</script>
